# 去除中文之间的空格，英文之间的空格保留
[参考资料](https://www.jb51.net/article/180134.htm)
```
def clean_space(text):
  """
  处理多余的空格
  """
  match_regex = re.compile(u'[\u4e00-\u9fa5。\.,，:：《》、\(\)（）]{1} +(?<![a-zA-Z])|\d+ +| +\d+|[a-z A-Z]+')
  should_replace_list = match_regex.findall(text)
  order_replace_list = sorted(should_replace_list,key=lambda i:len(i),reverse=True)
  for i in order_replace_list:
    if i == u' ':
      continue
    new_i = i.strip()
    text = text.replace(i,new_i)
  return text
  ```

# 中英混文本，对英文进行wordpiece分词
输入：
[uttid text] \
eg: 0001 今天的 weather 怎么样？\
中文和英文之前有空格，英文之间也有空格
```
import re
import sys
import sentencepiece as spm
sp = spm.SentencePieceProcessor()
sp.Load("bpe/bpe.model")  # 加载训练好的模型
test_text = "hello world"
test_text = test_text.split(' ')

for line in sys.stdin:
    line = line.strip()
    uttid = line.split(' ')[0]
    new_line = ''
    for w in line.split(' ')[1:]:
        if bool(re.search('[a-z]', w)):
            res = ' '.join(sp.EncodeAsPieces(w))  # 切词
            new_line += (res + ' ')
        else:
            new_line += (w + ' ')
    print(uttid + ' ' + new_line.rstrip())
```

# 统计数据库时长
输入是音频的路径
```
import wave
import contextlib
import os
from glob import glob
import sys

total_time = 0
count=0
for f in sys.stdin:
    f=f.strip()
    count +=1
    with contextlib.closing(wave.open(f, 'r')) as f:
        frames = f.getnframes()
    rate = f.getframerate()
    duration = frames / float(rate)
    total_time = total_time + duration
    # print("duration = ", duration)
print("total_time = ", total_time/3600)
print("总的语音段数 = ", count)
```
